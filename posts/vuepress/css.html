纯CSS扩展Vuepress功能

  • 标题、目录自动编号

标题、目录自动编号

将以下代码添加到.vuepress/styles/index.scss文件中：


// 全局标题自动编号
.vp-page {
  counter-reset: h2;

  h2 {
    counter-reset: h3;
    &:before {
      counter-increment: h2;
      content: counter(h2) ". ";
    }
  }

  h3 {
    counter-reset: h4;
    &:before {
      counter-increment: h3;
      content:  counter(h2) "." counter(h3) ". ";
    }
  }

  h4 {
    counter-reset: h5;
    &:before {
      counter-increment: h4;
      content:  counter(h2) "." counter(h3) "." counter(h4) ". ";
    }
  }

  h5 {
    counter-reset: h6;
    &:before {
      counter-increment: h5;
      content:  counter(h2) "." counter(h3) "." counter(h4) ". " counter(h5) ". ";
    }
  }
}
// 目录自动编号
.table-of-contents {
  counter-reset: hh2;

  ul>li>a {
    counter-reset: hh3;
    &:before {
      counter-increment: hh2;
      content: counter(hh2) ". ";
    }
  }

  ul>li>ul>li>a {
    counter-reset: hh4;
    &:before {
      counter-increment: hh3;
      content:  counter(hh2) "." counter(hh3) ". ";
    }
  }

  ul>li>ul>li>ul>li>a {
    counter-reset: hh5;
    &:before {
      counter-increment: hh4;
      content:  counter(hh3) "." counter(hh4) ". ";
    }
  }
}
